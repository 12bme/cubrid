USER 문

<create user>

CREATE USER <user name>
[PASSWORD password]
[GROUPS user name comma list]
[MEMBERS user name comma list]

<drop user>

DROP USER user name

<alter user>

ALTER USER user_name PASSWORD password

예제

CREATE USER david;

CREATE USER company;

CREATE USER engineering GROUPS company;

CREATE USER engineering MEMBERS david;

ALTER USER david PASSWORD 'passwd';

DROP USER david;
